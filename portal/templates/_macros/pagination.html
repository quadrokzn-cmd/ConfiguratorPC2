{# Этап 9А.2.3: пагинация по номерам страниц. Раньше в /admin/components была только пара «← назад / вперёд →» без номеров — на 28-страничной выдаче неудобно прыгать. Параметры: page — текущая страница (1-based); total_pages — всего страниц; href_for_page — макрос или callable, которому передаём номер страницы и получаем URL. Поскольку в Jinja передавать функции напрямую не очень принято, используем base_qs (строка query-string без &page=...) и сами добавляем '?...&page=N'. show_jump — показывать ли поле «перейти на страницу [ ]» (по умолчанию True для длинных списков). Логика выбора видимых страниц: - первая, последняя, текущая; - 1-2 страницы слева и справа от текущей; - между — «…», если разрыв >2. #} {% macro page_numbers(page, total_pages) -%} {# Возвращает упорядоченный список: либо номер страницы (int), либо None (для «…»). Хранится как сериализованный список — Jinja-макросы возвращают строку, поэтому генерим логику прямо в pagination_block. #} {%- endmacro %} {% macro pagination_block(page, total_pages, base_qs='', show_jump=True) -%} {%- if total_pages and total_pages > 1 -%} {%- set page = page|int -%} {%- set total_pages = total_pages|int -%} {%- set sep = '?' if not base_qs else (base_qs ~ '&') -%} {%- if base_qs and not (base_qs.startswith('?')) -%} {%- set sep = '?' ~ base_qs ~ '&' -%} {%- endif -%} {# Список номеров для показа: 1, last, page и ±2 от page. #} {%- set ns = namespace(prev=0) -%} {%- set keep = [] -%} {%- for n in range(1, total_pages + 1) -%} {%- if n == 1 or n == total_pages or (n >= page - 2 and n <= page + 2) -%} {%- set _ = keep.append(n) -%} {%- endif -%} {%- endfor -%}
{# ← назад #} {%- if page > 1 -%} ← {%- else -%} ← {%- endif -%} {# Числовые кнопки + '…' между разрывами. #} {%- for n in keep -%} {%- if ns.prev and n - ns.prev > 1 -%} … {%- endif -%} {%- if n == page -%} {{ n }} {%- else -%} {{ n }} {%- endif -%} {%- set ns.prev = n -%} {%- endfor -%} {# вперёд → #} {%- if page < total_pages -%} → {%- else -%} → {%- endif -%} {%- if show_jump and total_pages > 7 -%} перейти к {%- endif -%}
{%- endif -%} {%- endmacro %}
